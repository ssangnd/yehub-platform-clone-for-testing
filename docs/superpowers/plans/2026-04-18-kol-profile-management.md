# KOL Profile Management Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Implement KOL Category, KOL Tier, and KOL Profile management features (sections 5.3, 5.4, 5.5) with full backend CRUD APIs and frontend UI copied from yehub-demo.

**Architecture:** Backend-first approach. Prisma schema migration adds `Gender`/`LinkedBy` enums, `KolCategory`, `KolTier`, `ProfileCategory`, `ProfilePost` models, and extends `Profile`/`SocialAccount`. Three new NestJS modules (`kol-categories`, `kol-tiers`, `profiles`) provide REST APIs. Frontend copies demo pages into yehub-fe, replacing mock data with real API calls via TanStack Query.

**Tech Stack:** NestJS 11, Prisma 7, PostgreSQL 17, React 19, Vite, TanStack Query v5, shadcn/ui, Tailwind CSS v4

**Spec:** `docs/feature-kol-profile-management.md`

```typescript
import {
  ConflictException,
  Injectable,
  NotFoundException,
} from '@nestjs/common';
import { Prisma } from '../../../generated/prisma/client';
import { PrismaService } from '../prisma/prisma.service';
import { CreateKolCategoryDto } from './dto/create-kol-category.dto';
import { UpdateKolCategoryDto } from './dto/update-kol-category.dto';

@Injectable()
export class KolCategoriesService {
  constructor(private readonly prisma: PrismaService) {}

  async create(dto: CreateKolCategoryDto) {
    try {
      const category = await this.prisma.kolCategory.create({
        data: {
          name: dto.name,
          description: dto.description,
          color: dto.color ?? 'blue',
        },
      });
      return { ...category, profileCount: 0 };
    } catch (e) {
      if (
        e instanceof Prisma.PrismaClientKnownRequestError &&
        e.code === 'P2002'
      ) {
        throw new ConflictException('Category name already exists');
      }
      throw e;
    }
  }

  async findAll() {
    const categories = await this.prisma.kolCategory.findMany({
      orderBy: { name: 'asc' },
      include: { _count: { select: { profiles: true } } },
    });
    return categories.map((c) => ({
      id: c.id,
      name: c.name,
      description: c.description,
      color: c.color,
      profileCount: c._count.profiles,
      createdAt: c.created_at,
      updatedAt: c.updated_at,
    }));
  }

  async findOne(id: string) {
    const category = await this.prisma.kolCategory.findUnique({
      where: { id },
      include: { _count: { select: { profiles: true } } },
    });
    if (!category) throw new NotFoundException('Category not found');
    return {
      id: category.id,
      name: category.name,
      description: category.description,
      color: category.color,
      profileCount: category._count.profiles,
      createdAt: category.created_at,
      updatedAt: category.updated_at,
    };
  }

  async update(id: string, dto: UpdateKolCategoryDto) {
    try {
      const category = await this.prisma.kolCategory.update({
        where: { id },
        data: dto,
        include: { _count: { select: { profiles: true } } },
      });
      return {
        id: category.id,
        name: category.name,
        description: category.description,
        color: category.color,
        profileCount: category._count.profiles,
        createdAt: category.created_at,
        updatedAt: category.updated_at,
      };
    } catch (e) {
      if (
        e instanceof Prisma.PrismaClientKnownRequestError &&
        e.code === 'P2025'
      ) {
        throw new NotFoundException('Category not found');
      }
      if (
        e instanceof Prisma.PrismaClientKnownRequestError &&
        e.code === 'P2002'
      ) {
        throw new ConflictException('Category name already exists');
      }
      throw e;
    }
  }

  async remove(id: string) {
    try {
      await this.prisma.kolCategory.delete({ where: { id } });
    } catch (e) {
      if (
        e instanceof Prisma.PrismaClientKnownRequestError &&
        e.code === 'P2025'
      ) {
        throw new NotFoundException('Category not found');
      }
      throw e;
    }
  }
}
```

```typescript
import {
  ConflictException,
  Injectable,
  NotFoundException,
} from '@nestjs/common';
import { Prisma } from '../../../generated/prisma/client';
import { PrismaService } from '../prisma/prisma.service';
import { CreateKolTierDto } from './dto/create-kol-tier.dto';
import { UpdateKolTierDto } from './dto/update-kol-tier.dto';

@Injectable()
export class KolTiersService {
  constructor(private readonly prisma: PrismaService) {}

  private formatTier(
    tier: any & { _count: { profiles: number } },
  ) {
    return {
      id: tier.id,
      name: tier.name,
      description: tier.description,
      color: tier.color,
      minFollowers: tier.min_followers,
      maxFollowers: tier.max_followers,
      profileCount: tier._count.profiles,
      createdAt: tier.created_at,
      updatedAt: tier.updated_at,
    };
  }

  async create(dto: CreateKolTierDto) {
    try {
      const tier = await this.prisma.kolTier.create({
        data: {
          name: dto.name,
          description: dto.description,
          color: dto.color ?? 'blue',
          min_followers: dto.minFollowers ?? 0,
          max_followers: dto.maxFollowers ?? null,
        },
        include: { _count: { select: { profiles: true } } },
      });
      return this.formatTier(tier);
    } catch (e) {
      if (
        e instanceof Prisma.PrismaClientKnownRequestError &&
        e.code === 'P2002'
      ) {
        throw new ConflictException('Tier name already exists');
      }
      throw e;
    }
  }

  async findAll() {
    const tiers = await this.prisma.kolTier.findMany({
      orderBy: { min_followers: 'desc' },
      include: { _count: { select: { profiles: true } } },
    });
    return tiers.map((t) => this.formatTier(t));
  }

  async findOne(id: string) {
    const tier = await this.prisma.kolTier.findUnique({
      where: { id },
      include: { _count: { select: { profiles: true } } },
    });
    if (!tier) throw new NotFoundException('Tier not found');
    return this.formatTier(tier);
  }

  async update(id: string, dto: UpdateKolTierDto) {
    try {
      const data: Prisma.KolTierUpdateInput = {};
      if (dto.name !== undefined) data.name = dto.name;
      if (dto.description !== undefined) data.description = dto.description;
      if (dto.color !== undefined) data.color = dto.color;
      if (dto.minFollowers !== undefined) data.min_followers = dto.minFollowers;
      if (dto.maxFollowers !== undefined) data.max_followers = dto.maxFollowers;

      const tier = await this.prisma.kolTier.update({
        where: { id },
        data,
        include: { _count: { select: { profiles: true } } },
      });
      return this.formatTier(tier);
    } catch (e) {
      if (
        e instanceof Prisma.PrismaClientKnownRequestError &&
        e.code === 'P2025'
      ) {
        throw new NotFoundException('Tier not found');
      }
      if (
        e instanceof Prisma.PrismaClientKnownRequestError &&
        e.code === 'P2002'
      ) {
        throw new ConflictException('Tier name already exists');
      }
      throw e;
    }
  }

  async remove(id: string) {
    try {
      await this.prisma.kolTier.delete({ where: { id } });
    } catch (e) {
      if (
        e instanceof Prisma.PrismaClientKnownRequestError &&
        e.code === 'P2025'
      ) {
        throw new NotFoundException('Tier not found');
      }
      throw e;
    }
  }
}
```

```typescript
import {
  ConflictException,
  Injectable,
  NotFoundException,
} from '@nestjs/common';
import { Prisma } from '../../../generated/prisma/client';
import { PrismaService } from '../prisma/prisma.service';
import { CreateProfileDto } from './dto/create-profile.dto';
import { LinkAccountDto } from './dto/link-account.dto';
import { LinkPostDto } from './dto/link-post.dto';
import { ListProfilesQueryDto } from './dto/list-profiles-query.dto';
import { MoveAccountDto } from './dto/move-account.dto';
import { UpdateProfileDto } from './dto/update-profile.dto';

@Injectable()
export class ProfilesService {
  constructor(private readonly prisma: PrismaService) {}

  private readonly profileInclude = {
    tier: { select: { id: true, name: true, color: true } },
    categories: {
      include: {
        kolCategory: { select: { id: true, name: true, color: true } },
      },
    },
    socialAccounts: {
      orderBy: { follower_count: 'desc' as const },
    },
    _count: { select: { profilePosts: true } },
  };

  private formatProfile(p: any) {
    const accounts = (p.socialAccounts ?? []).map((a: any) => ({
      id: a.id,
      platform: a.platform,
      platformUserId: a.platform_user_id,
      username: a.username,
      displayName: a.display_name,
      followerCount: a.follower_count,
      isVerified: a.is_verified,
      avatarUrl: a.avatar_url,
      createdAt: a.created_at,
    }));

    return {
      id: p.id,
      name: p.name,
      description: p.description,
      gender: p.gender,
      email: p.email,
      phone: p.phone,
      tags: p.tags,
      tier: p.tier
        ? { id: p.tier.id, name: p.tier.name, color: p.tier.color }
        : null,
      categories: (p.categories ?? []).map((pc: any) => ({
        id: pc.kolCategory.id,
        name: pc.kolCategory.name,
        color: pc.kolCategory.color,
      })),
      totalFollowers: accounts.reduce(
        (sum: number, a: any) => sum + a.followerCount,
        0,
      ),
      accounts,
      linkedPostCount: p._count?.profilePosts ?? 0,
      createdAt: p.created_at,
      updatedAt: p.updated_at,
    };
  }

  private extractUsernameFromUrl(url: string): string {
    const cleaned = url.replace(/\/+$/, '');
    const parts = cleaned.split('/');
    const last = parts[parts.length - 1];
    return last.replace(/^@/, '');
  }

  async create(dto: CreateProfileDto) {
    const profile = await this.prisma.profile.create({
      data: {
        name: dto.name,
        description: dto.description,
        gender: dto.gender,
        email: dto.email,
        phone: dto.phone,
        tags: dto.tags ?? [],
        tier_id: dto.tierId,
        categories: dto.categoryIds?.length
          ? {
              create: dto.categoryIds.map((id) => ({
                kol_category_id: id,
              })),
            }
          : undefined,
        socialAccounts: dto.socialAccounts?.length
          ? {
              create: dto.socialAccounts.map((sa) => {
                const username = this.extractUsernameFromUrl(sa.url);
                return {
                  platform: sa.platform,
                  platform_user_id: `${sa.platform.toLowerCase()}_${username}`,
                  username,
                  display_name: dto.name,
                };
              }),
            }
          : undefined,
      },
      include: this.profileInclude,
    });
    return this.formatProfile(profile);
  }

  async findAll(query: ListProfilesQueryDto) {
    const page = query.page ?? 1;
    const limit = query.limit ?? 20;
    const skip = (page - 1) * limit;

    const where: Prisma.ProfileWhereInput = {};
    const conditions: Prisma.ProfileWhereInput[] = [];

    if (query.search) {
      const search = query.search.trim();
      conditions.push({
        OR: [
          { name: { contains: search, mode: 'insensitive' } },
          { tags: { hasSome: [search] } },
          {
            socialAccounts: {
              some: {
                username: { contains: search, mode: 'insensitive' },
              },
            },
          },
        ],
      });
    }

    if (query.categoryIds) {
      const ids = query.categoryIds.split(',').filter(Boolean);
      if (ids.length) {
        conditions.push({
          categories: { some: { kol_category_id: { in: ids } } },
        });
      }
    }

    if (query.tierIds) {
      const ids = query.tierIds.split(',').filter(Boolean);
      if (ids.length) {
        conditions.push({ tier_id: { in: ids } });
      }
    }

    if (query.platforms) {
      const platforms = query.platforms.split(',').filter(Boolean);
      if (platforms.length) {
        conditions.push({
          socialAccounts: {
            some: { platform: { in: platforms as any } },
          },
        });
      }
    }

    if (query.genders) {
      const genders = query.genders.split(',').filter(Boolean);
      if (genders.length) {
        conditions.push({ gender: { in: genders as any } });
      }
    }

    if (conditions.length) {
      where.AND = conditions;
    }

    const orderBy = this.buildOrderBy(query.sortBy, query.sortOrder);

    const [profiles, total] = await this.prisma.$transaction([
      this.prisma.profile.findMany({
        where,
        include: this.profileInclude,
        orderBy,
        skip,
        take: limit,
      }),
      this.prisma.profile.count({ where }),
    ]);

    return {
      data: profiles.map((p) => this.formatProfile(p)),
      meta: {
        total,
        page,
        limit,
        totalPages: Math.ceil(total / limit) || 1,
      },
    };
  }

  private buildOrderBy(
    sortBy?: string,
    sortOrder?: string,
  ): Prisma.ProfileOrderByWithRelationInput {
    const dir = (sortOrder === 'asc' ? 'asc' : 'desc') as Prisma.SortOrder;
    switch (sortBy) {
      case 'name':
        return { name: dir };
      case 'totalFollowers':
        return { socialAccounts: { _count: dir } };
      case 'linkedPostCount':
        return { profilePosts: { _count: dir } };
      default:
        return { created_at: dir };
    }
  }

  async findOne(id: string) {
    const profile = await this.prisma.profile.findUnique({
      where: { id },
      include: {
        ...this.profileInclude,
        profilePosts: {
          include: {
            post: {
              select: {
                id: true,
                url: true,
                platform: true,
                campaign: { select: { id: true, name: true } },
              },
            },
          },
          orderBy: { created_at: 'desc' },
        },
      },
    });
    if (!profile) throw new NotFoundException('Profile not found');

    const formatted = this.formatProfile(profile);
    return {
      ...formatted,
      linkedPosts: profile.profilePosts.map((pp) => ({
        id: pp.post.id,
        url: pp.post.url,
        platform: pp.post.platform,
        campaignName: pp.post.campaign.name,
        linkedBy: pp.linked_by,
        createdAt: pp.created_at,
      })),
    };
  }

  async update(id: string, dto: UpdateProfileDto) {
    const existing = await this.prisma.profile.findUnique({ where: { id } });
    if (!existing) throw new NotFoundException('Profile not found');

    const data: Prisma.ProfileUpdateInput = {};
    if (dto.name !== undefined) data.name = dto.name;
    if (dto.description !== undefined) data.description = dto.description;
    if (dto.gender !== undefined) data.gender = dto.gender;
    if (dto.email !== undefined) data.email = dto.email;
    if (dto.phone !== undefined) data.phone = dto.phone;
    if (dto.tags !== undefined) data.tags = dto.tags;
    if (dto.tierId !== undefined) {
      data.tier = dto.tierId
        ? { connect: { id: dto.tierId } }
        : { disconnect: true };
    }

    if (dto.categoryIds !== undefined) {
      await this.prisma.profileCategory.deleteMany({
        where: { profile_id: id },
      });
      if (dto.categoryIds.length) {
        await this.prisma.profileCategory.createMany({
          data: dto.categoryIds.map((catId) => ({
            profile_id: id,
            kol_category_id: catId,
          })),
        });
      }
    }

    const profile = await this.prisma.profile.update({
      where: { id },
      data,
      include: this.profileInclude,
    });
    return this.formatProfile(profile);
  }

  async remove(id: string) {
    try {
      await this.prisma.profile.delete({ where: { id } });
    } catch (e) {
      if (
        e instanceof Prisma.PrismaClientKnownRequestError &&
        e.code === 'P2025'
      ) {
        throw new NotFoundException('Profile not found');
      }
      throw e;
    }
  }

  async linkAccount(profileId: string, dto: LinkAccountDto) {
    const profile = await this.prisma.profile.findUnique({
      where: { id: profileId },
    });
    if (!profile) throw new NotFoundException('Profile not found');

    try {
      const account = await this.prisma.socialAccount.create({
        data: {
          profile_id: profileId,
          platform: dto.platform,
          platform_user_id:
            dto.platformUserId ??
            `${dto.platform.toLowerCase()}_${dto.username}`,
          username: dto.username,
          display_name: dto.displayName ?? profile.name,
          avatar_url: dto.avatarUrl,
        },
      });
      return {
        id: account.id,
        platform: account.platform,
        platformUserId: account.platform_user_id,
        username: account.username,
        displayName: account.display_name,
        followerCount: account.follower_count,
        isVerified: account.is_verified,
        avatarUrl: account.avatar_url,
        createdAt: account.created_at,
      };
    } catch (e) {
      if (
        e instanceof Prisma.PrismaClientKnownRequestError &&
        e.code === 'P2002'
      ) {
        throw new ConflictException(
          'This account is already linked to a profile',
        );
      }
      throw e;
    }
  }

  async unlinkAccount(profileId: string, accountId: string) {
    const account = await this.prisma.socialAccount.findFirst({
      where: { id: accountId, profile_id: profileId },
    });
    if (!account)
      throw new NotFoundException('Social account not found on this profile');

    await this.prisma.socialAccount.delete({ where: { id: accountId } });
  }

  async moveAccount(
    profileId: string,
    accountId: string,
    dto: MoveAccountDto,
  ) {
    const account = await this.prisma.socialAccount.findFirst({
      where: { id: accountId, profile_id: profileId },
    });
    if (!account)
      throw new NotFoundException('Social account not found on this profile');

    const targetProfile = await this.prisma.profile.findUnique({
      where: { id: dto.targetProfileId },
    });
    if (!targetProfile)
      throw new NotFoundException('Target profile not found');

    await this.prisma.socialAccount.update({
      where: { id: accountId },
      data: { profile_id: dto.targetProfileId },
    });
  }

  async linkPost(profileId: string, dto: LinkPostDto) {
    const profile = await this.prisma.profile.findUnique({
      where: { id: profileId },
    });
    if (!profile) throw new NotFoundException('Profile not found');

    const post = await this.prisma.post.findUnique({
      where: { id: dto.postId },
    });
    if (!post) throw new NotFoundException('Post not found');

    try {
      await this.prisma.profilePost.create({
        data: {
          profile_id: profileId,
          post_id: dto.postId,
          linked_by: 'MANUAL',
        },
      });
    } catch (e) {
      if (
        e instanceof Prisma.PrismaClientKnownRequestError &&
        e.code === 'P2002'
      ) {
        throw new ConflictException(
          'This post is already linked to this profile',
        );
      }
      throw e;
    }
  }

  async unlinkPost(profileId: string, postId: string) {
    try {
      await this.prisma.profilePost.delete({
        where: {
          profile_id_post_id: { profile_id: profileId, post_id: postId },
        },
      });
    } catch (e) {
      if (
        e instanceof Prisma.PrismaClientKnownRequestError &&
        e.code === 'P2025'
      ) {
        throw new NotFoundException('Post link not found');
      }
      throw e;
    }
  }
}
```

```typescript
import { useState } from 'react'
import { useQuery } from '@tanstack/react-query'
import { useDebounce } from '@/hooks/use-debounce'
import { queryKeys } from '@/lib/constants/query-keys'
import { profilesApi, type ListProfilesParams } from '@/api/profiles'
import { kolCategoriesApi } from '@/api/kol-categories'
import { kolTiersApi } from '@/api/kol-tiers'

export function useProfilesList() {
  const [page, setPage] = useState(1)
  const [search, setSearch] = useState('')
  const [sortBy, setSortBy] = useState<string | undefined>(undefined)
  const [sortOrder, setSortOrder] = useState<'asc' | 'desc'>('desc')
  const [filters, setFilters] = useState<{
    categoryIds?: string
    tierIds?: string
    platforms?: string
    genders?: string
  }>({})

  const debouncedSearch = useDebounce(search, 300)

  const params: ListProfilesParams = {
    search: debouncedSearch || undefined,
    page,
    limit: 20,
    sortBy,
    sortOrder,
    ...filters,
  }

  const profilesQuery = useQuery({
    queryKey: queryKeys.profiles.list(params as Record<string, unknown>),
    queryFn: () => profilesApi.list(params),
  })

  const categoriesQuery = useQuery({
    queryKey: queryKeys.kolCategories,
    queryFn: kolCategoriesApi.list,
  })

  const tiersQuery = useQuery({
    queryKey: queryKeys.kolTiers,
    queryFn: kolTiersApi.list,
  })

  const handleSort = (key: string) => {
    if (sortBy === key) {
      setSortOrder((prev) => (prev === 'asc' ? 'desc' : 'asc'))
    } else {
      setSortBy(key)
      setSortOrder('desc')
    }
    setPage(1)
  }

  return {
    profiles: profilesQuery.data?.data ?? [],
    meta: profilesQuery.data?.meta,
    isLoading: profilesQuery.isLoading,
    categories: categoriesQuery.data ?? [],
    tiers: tiersQuery.data ?? [],
    search,
    setSearch,
    page,
    setPage,
    sortBy,
    sortOrder,
    handleSort,
    filters,
    setFilters,
  }
}
```

```typescript
    queryKey: queryKeys.kolTiers,
    queryFn: kolTiersApi.list,
  })

  const invalidate = () => {
    queryClient.invalidateQueries({ queryKey: queryKeys.profile(id!) })
    queryClient.invalidateQueries({ queryKey: queryKeys.profiles.all })
  }

  const updateMutation = useMutation({
    mutationFn: (data: UpdateProfilePayload) => profilesApi.update(id!, data),
    onSuccess: () => { invalidate(); toast.success('Profile updated') },
  })

  const linkAccountMutation = useMutation({
    mutationFn: (data: LinkAccountPayload) => profilesApi.linkAccount(id!, data),
    onSuccess: () => { invalidate(); toast.success('Account linked') },
  })

  const unlinkAccountMutation = useMutation({
    mutationFn: (accountId: string) => profilesApi.unlinkAccount(id!, accountId),
    onSuccess: () => { invalidate(); toast.success('Account unlinked') },
  })

  const moveAccountMutation = useMutation({
    mutationFn: ({ accountId, targetProfileId }: { accountId: string; targetProfileId: string }) =>
      profilesApi.moveAccount(id!, accountId, targetProfileId),
    onSuccess: () => { invalidate(); toast.success('Account moved') },
  })

  const linkPostMutation = useMutation({
    mutationFn: (postId: string) => profilesApi.linkPost(id!, postId),
    onSuccess: () => { invalidate(); toast.success('Post linked') },
  })

  const unlinkPostMutation = useMutation({
    mutationFn: (postId: string) => profilesApi.unlinkPost(id!, postId),
    onSuccess: () => { invalidate(); toast.success('Post unlinked') },
  })

  return {
    profile: profileQuery.data,
    isLoading: profileQuery.isLoading,
    categories: categoriesQuery.data ?? [],
    tiers: tiersQuery.data ?? [],
    updateProfile: updateMutation.mutate,
    linkAccount: linkAccountMutation.mutate,
    unlinkAccount: unlinkAccountMutation.mutate,
    moveAccount: moveAccountMutation.mutate,
    linkPost: linkPostMutation.mutate,
    unlinkPost: unlinkPostMutation.mutate,
  }
}
```
